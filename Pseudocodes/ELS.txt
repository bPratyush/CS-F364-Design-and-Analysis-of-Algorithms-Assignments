proc BronKerboschDegeneracy(V, E)
1: for each vertex vi in a degeneracy ordering v0, v1, v2, . . . of (V;E) do
2: P <- gamma(Vi) intersection {vi+1,....,vn-1}
3: X <- gamma(Vi) intersection {v0,....,vi-1}
4: BronKerboschPivot(P, fvig, X)
5: end for

proc BronKerboschPivot(P, R, X)
1: if P union X = empty then
2: report R as a maximal clique
3: end if
4: choose a pivot u belonging to P union X
5: for each vertex v belonging to P \ gamma(u) do
6: BronKerboschPivot(P intersection gamma(v), R union {v}, X intersection gamma(v))
7: P <- P \ {v}
8: X <- X union {v}
9: end for
