procedure EXPAND(SUBG,CAND)
begin
    if SUBG=empty
        then print ("clique") /* to represent Q is maximal clique */
    else u:= a vertex u in SUBG that maximises |CAND intersection gamma(u)|;
        /* let EXTu = CAND - gamma(u); */
        /* FINI := empty */
        while CAND-gamma(u)!=empty
            do q:= a vertex in (CAND-gamma(u));
            print(q,",");
    /* to represent next statement */
    /*        Q:=Q union {q}; */
              SUBGq = SUBG intersection gamma(q);
              CANDq = CAND intersection gamma(q);
              EXPAND(SUBGq,CANDq);
              CAND:=CAND-{q}; /*FINI:=FINI union {q};*/
              print("back,");
    /* to represent next statement */
    /*        Q:=Q - {q}; */
        od
    fi
end of EXPAND
